<!-- Additional fields required by HA Core hassfest: iot_class, integration_type, config_flow -->

Required (HACS + hassfest combined):
  domain: str (package directory name)
  name: str
  version: str (AwesomeVersion parseable)
  documentation: url
  issue_tracker: url
  codeowners: list[str]
  iot_class: one of {assumed_state, cloud_polling, cloud_push, local_polling, local_push, calculated}
  integration_type: one of {device, hub, service, entity, hardware, helper, system, virtual}
  config_flow: bool (if true, config_flow.py must exist — Phase 1 sets false)

Optional:
  requirements: list[str] (pip deps; Phase 1 uses [])
  dependencies: list[str] (HA integration deps; Phase 1 uses [])
  after_dependencies: list[str]

async def async_unload_entry(hass: HomeAssistant, entry: ConfigEntry) -> bool
